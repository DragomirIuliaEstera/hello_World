public class ContBancar {
    private double sold = 0;

    public double getSold() {
        return sold;
    }

    public ContBancar(double sold){
        this.sold = sold;
    }

    public void depozitareSuma(double sum) {
        this.sold = this.sold + sum;
    }
//  Ex2
    public boolean retragereSuma(int sumaRetrasa){
        if(sumaRetrasa <= this.sold) {
            this.sold = this.sold - sumaRetrasa;
            return true;
        }
        else {
            return false;
        }
    }

    public void interogareSold(){
        System.out.println("Soldul este "+this.sold+ " lei.");
    }

}

public class car{
    private String _type;
    private String _colour;
    private double _speed;

    public car(String type, String colour, double speed){
        _type = type;
        _colour = colour;
        _speed = speed;
    }

    public String get_type() {
        return _type;
    }

    public void set_type(String _type) {
        this._type = _type;
    }

    public String get_colour() {
        return _colour;
    }

    public void set_colour(String _colour) {
        this._colour = _colour;
    }

    public double get_speed() {
        return _speed;
    }

    public void set_speed(double _speed) {
        this._speed = _speed;
    }

    public String toString(){
        String res = "Informatiile despre masina sunt: ";
        res += _type;
        res += "; "+ _colour;
        res += "; "+ _speed;

        return res;
    }
}
